# Check This
- `double x, y` for real number operation
- `cout<<fixed; cout.precision(1);` for make round to one decimal place

```cpp
#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int n;
double x, y;

struct Info {
	double x;
	double y;
};

Info node[10001];

int main()
{
	cin >> n;

  // MAKE NODE
  /* UPDATE NODE */
	for (int i = 0; i < n; i++) {
		cin >> x >> y;
		node[i].x = x;
		node[i].y = y;
	}
  /* ADD first node AT LAST */
	node[n].x = node[0].x;
	node[n].y = node[0].y;

  // MAKE ANSWER
  double ans = 0;
  /* ADD CROSS PRODUCT */
	for (int i = 0; i < n; i++) {
		ans += (node[i].x * node[i + 1].y);
		ans -= (node[i + 1].x * node[i].y);
	}
  /* DIVIDE by 2 */
	ans = abs(ans / 2.0);

  // PRINT ANSWER
	cout << fixed;
	cout.precision(1);
	cout << ans;

	return 0;
}
```
